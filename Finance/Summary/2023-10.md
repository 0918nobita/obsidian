```dataview
TABLE WITHOUT ID
    choice(contains(tags, "支出"), amount * -1, amount) as income

```
### 支出
```dataview
TABLE WITHOUT ID
    file.name AS "費目",
    date.day AS "日付",
    amount AS "金額"
FROM #支出
WHERE date.year = 2023 AND date.month = 10
SORT date ASC
```
```dataview
TABLE WITHOUT ID
    sum(rows.amount) AS "合計"
FROM #支出
WHERE date.year = 2023 AND date.month = 10
GROUP BY true
```
